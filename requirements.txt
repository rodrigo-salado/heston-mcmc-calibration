# ---------------------------------------------------------------------------
# Core Numerical Computing
# ---------------------------------------------------------------------------

# NumPy: Fundamental package for scientific computing in Python.
# Used for: Array operations, vectorised Fourier integration,
#           random number generation for MCMC proposals,
#           statistical calculations (mean, std, quantiles),
#           and all matrix/vector operations throughout the calibration.
numpy>=1.24.0,<2.0.0

# SciPy: Library for scientific computing and optimisation.
# Used for: Statistical distributions (norm for likelihood and priors),
#           Brent's method for implied volatility root-finding,
#           numerical integration utilities, and special functions.
scipy>=1.10.0

# ---------------------------------------------------------------------------
# Data Manipulation and Analysis
# ---------------------------------------------------------------------------

# Pandas: Data manipulation and analysis library.
# Used for: Organising calibration results into DataFrames,
#           handling multi-chain MCMC output, summary statistics,
#           and exporting results to CSV/Excel for further analysis.
pandas>=2.0.0

# ---------------------------------------------------------------------------
# Data Visualization
# ---------------------------------------------------------------------------

# Matplotlib: Comprehensive plotting library for Python.
# Used for: Generating all visualizations including:
#           - Trace plots for MCMC convergence diagnostics
#           - Marginal posterior density plots
#           - Pair plots with correlation coefficients
#           - Posterior predictive smile validation plots
matplotlib>=3.7.0

# Seaborn: Statistical data visualization library.
# Used for: Enhanced pair plots with KDE on diagonals,
#           beautiful default styling, correlation heatmaps,
#           and joint distribution visualisations.
seaborn>=0.12.0

# ---------------------------------------------------------------------------
# MCMC and Bayesian Inference (Optional - for comparison)
# ---------------------------------------------------------------------------

# PyMC: Probabilistic programming library with NUTS sampler.
# Used for: Cross-validation of our custom Metropolis-Hastings
#           implementation, automatic tuning, and gradient-based
#           sampling for comparison studies.
# Note: Optional dependency - not required for main calibration
pymc>=5.0.0

# ArviZ: Exploratory analysis of Bayesian models.
# Used for: Computing R-hat convergence statistics,
#           effective sample size (ESS) calculations,
#           and generating diagnostic plots for PyMC results.
arviz>=0.15.0

# ---------------------------------------------------------------------------
# Development and Testing
# ---------------------------------------------------------------------------

# pytest: Mature testing framework for Python.
# Used for: Unit tests for all calibration components:
#           - Heston characteristic function correctness
#           - BS limit verification (xi -> 0)
#           - Fourier inversion accuracy
#           - MCMC acceptance rate validation
#           - Prior and likelihood evaluation
pytest>=7.4.0

# pytest-cov: Coverage reporting plugin for pytest.
# Used for: Measuring test coverage to ensure critical paths
#           (pricing, likelihood, MCMC proposals) are fully tested.
pytest-cov>=4.1.0

# pytest-benchmark: Performance benchmarking plugin for pytest.
# Used for: Measuring and tracking performance of critical operations:
#           - Likelihood evaluation speed
#           - Posterior evaluation speed
#           - Single MCMC iteration time
#           - Full calibration runtime
#           - Fourier integration performance
# Note: Optional - for performance regression testing.
# Usage: pytest tests/ --benchmark-only
#        pytest tests/ --benchmark-compare=previous.json
pytest-benchmark>=4.0.0

#   pytest tests/test_inference.py -k "benchmark" --benchmark-only
#
# To compare benchmark results across runs:
#   pytest tests/test_inference.py --benchmark-autosave
